CREATE TABLE "Ages" (
	"ID"	INTEGER UNIQUE,
	"WeeksAge"	INTEGER NOT NULL,
	"Age"	TEXT,
	PRIMARY KEY("ID" AUTOINCREMENT)
)

CREATE TABLE "Areas" (
	"ID"	INTEGER UNIQUE,
	"Name"	TEXT NOT NULL,
	PRIMARY KEY("ID" AUTOINCREMENT)
)

CREATE TABLE "Behaviors" (
	"ID"	INTEGER UNIQUE,
	"Age_ID"	INTEGER NOT NULL,
	"Area_ID"	INTEGER NOT NULL,
	"Description"	TEXT NOT NULL,
	PRIMARY KEY("ID" AUTOINCREMENT),
	FOREIGN KEY("Age_ID") REFERENCES "Ages"("ID"),
	FOREIGN KEY("Area_ID") REFERENCES "Areas"("ID")
)

CREATE TABLE "Kids" (
	"ID"	INTEGER UNIQUE,
	"Name"	TEXT NOT NULL,
	"Gender"	TEXT,
	"DayBirth"	DATE NOT NULL,
	"Age_ID"	INTEGER NOT NULL,
	"Address"	TEXT NOT NULL,
	"BloodType"	TEXT,
	"Allergic"	INTEGER DEFAULT 0,
	"WhatAllergy"	TEXT,
	"Status"	INTEGER NOT NULL DEFAULT 1,
	PRIMARY KEY("ID" AUTOINCREMENT),
	FOREIGN KEY("Age_ID") REFERENCES "Ages"("ID")
)

CREATE TABLE "Registers" (
	"Kid_ID"	INTEGER NOT NULL,
	"Age_ID"	INTEGER NOT NULL,
	"Area_ID"	INTEGER NOT NULL,
	"Behavior_ID"	INTEGER NOT NULL,
	"Status"	INTEGER NOT NULL,
	"Comments"	TEXT,
	"RecordDate"	DATETIME DEFAULT (DATETIME('now', '-5 hours')),
	FOREIGN KEY("Age_ID") REFERENCES "Ages"("ID"),
	FOREIGN KEY("Area_ID") REFERENCES "Areas"("ID"),
	FOREIGN KEY("Behavior_ID") REFERENCES "Behaviors"("ID"),
	FOREIGN KEY("Kid_ID") REFERENCES "Kids"("ID")
);

CREATE TABLE "Evaluations" (
	"ID"	INTEGER UNIQUE,
	"Kid_ID"	INTEGER NOT NULL,
	"Age_ID"	INTEGER NOT NULL,
	"Adaptative" INTEGER NOT NULL,
	"GrossMotor" INTEGER NOT NULL,
	"FineMotor" INTEGER NOT NULL,
	"Language" INTEGER NOT NULL,
	"SocialPerson" INTEGER NOT NULL,
	"Comments"	TEXT,
	"WeekAttention" TEXT,
	"RecordDate"	DATETIME DEFAULT (DATETIME('now', '-5 hours')),
	PRIMARY KEY("ID" AUTOINCREMENT),
	FOREIGN KEY("Kid_ID") REFERENCES "Kids"("ID"),
	FOREIGN KEY("Age_ID") REFERENCES "Ages"("ID")
);

CREATE TABLE "EvaluationsDetail" (
	"Evaluation_ID"	INTEGER NOT NULL,
	"Area_ID"	INTEGER NOT NULL,
	"Behavior_ID"	INTEGER NOT NULL,
	"Status"	INTEGER NOT NULL,
	FOREIGN KEY("Evaluation_ID") REFERENCES "Evaluations"("ID"),
	FOREIGN KEY("Area_ID") REFERENCES "Areas"("ID"),
	FOREIGN KEY("Behavior_ID") REFERENCES "Behaviors"("ID")
);
